          if(arr.length<=1){
        return arr;

       function subble(arr){
         if(arr.length<=1){
           return arr;
        } 
        else{
          for(var i=0;i<arr.length;i++){
            for(var j=0;j<arr.length-i;j++){
             if(arr[j]>arr[j+1]){
               var temp=arr[j];
                  arr[j]=arr[j+1];
                  arr[j+1]=temp;
              }
            }
           }
        return arr;
        }
      }
		alert(subble([1,234,45,34]));

**快速排序：不稳定，时间复杂度 最理想 O(nlogn) 最差时间O(n^2)**

                  function quick(arr){
				if(arr.length<=1){
					return arr;
				}
				else{
				var left=[];
				var right=[];
				var middle=arr[0];
				for(var i=1;i<arr.length;i++){
					if(arr[i]<middle){
						left.push(arr[i]);
					}
					else{
						right.push(arr[i]);
					}
				}
				return quick(left).concat([middle],quick(right));
				}
			}
			alert(quick([1,23,34,2,3,5,2]));

                            function two(arr,x){
				var low=0;
				var high=arr.length-1;
				while(low<=high){
					var middle=Math.floor((low+high)/2);
					if(x==arr[middle]){
						return middle;
					}
					if(x<arr[middle]){
						high=middle-1;
					}
					else{
						low=middle+1;
					}
				}
				return -1;
			}
